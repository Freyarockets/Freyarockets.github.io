func open_all_links():

        "https://Freyarockets.github.io",
        "https://app.uniswap.org/#/swap?inputCurrency=ETH&outputCurrency=0xE3240d094fe549E365033688c9DeBe645bA44db6&exactAmount=0.1",

    ]

    for url in urls:
        var error = OS.shell_open(url)
        if error != OK:
            print("Failed to open URL: ", url, " Error code: ", error)
        else:
            print("Attempting to open: ", url)
